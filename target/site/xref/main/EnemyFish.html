1   package main;
2   
3   import java.util.Random;
4   
5   import javafx.scene.image.Image;
6   
7   /**
8    * An enemy fish is any fish other than the player fish. It inherits
9    * the movement speed and sprite attributes from the general entity class.
10   * The added boolean 'isLefty' simply indicates if the fish spawns on the left side of the screen.
11   * If 'isLefty' is false, the fish spawns at the right side of the screen.
12   * @author Clinton Cao, Dmity Malarev, Matthijs Halvemaan, Sunwei Wang, Michiel Doesburg.
13   */
14  public class EnemyFish extends Entity {
15  	private static String leftImageFileName = "EnemyFish_Left.png";
16  	private static String rightImageFileName = "EnemyFish_Right.png";
17  	
18  	private boolean isLefty;
19  	
20  	public EnemyFish(int movespeed, boolean isLefty, Sprite sprite) {
21  		super(movespeed, sprite);
22  		this.isLefty = isLefty;
23  	}
24  	
25  	/** Generate an enemy fish to be placed on the screen.
26  	 * Make a new fish with a random movement speed, random height at which it spawns, and random
27  	 * side of the screen at which it spawns.
28  	 * @return an enemy fish.
29  	 */
30  	public static EnemyFish generateFish() {
31  		
32  		Random rng = new Random();
33  		
34  		// Generate the height at which the fish spawns, its movement speed, and at which side of the screen it spawns.
35  		int randomHeight = rng.nextInt(600);
36  		int randomSpeed = rng.nextInt(9) + 1;
37  		boolean isLefty = rng.nextBoolean();
38  		
39  		// Generate the fish, depending on which side of the screen it spawns.
40  		if(isLefty) {
41  			// Get the image for the fish and its respective height and width.
42  			Image fishImage = new Image(rightImageFileName);
43  			int fishImageWidth = (int) fishImage.getWidth();
44  			int fishImageHeight = (int) fishImage.getHeight();
45  			
46  			return new EnemyFish(randomSpeed, isLefty, new Sprite(fishImage, new AABB(-fishImageWidth, randomHeight, fishImageWidth, fishImageHeight)));
47  		} else {
48  			Image fishImage = new Image(leftImageFileName);
49  			int fishImageWidth = (int) fishImage.getWidth();
50  			int fishImageHeight = (int) fishImage.getHeight();
51  						
52  			// If the fish spawns at the right side of the screen, it needs to be placed at the X coordinate equal to the width of the AABB screenbox;
53  			return new EnemyFish(randomSpeed, isLefty, new Sprite(fishImage, new AABB(MainScreenController.getScreenbox().getWidth(), randomHeight, fishImageWidth, fishImageHeight)));
54  		}
55  	}
56  
57  	public boolean isLefty() {
58  		return isLefty;
59  	}
60  
61  	public void setLefty(boolean isLefty) {
62  		this.isLefty = isLefty;
63  	}
64  }

15  	public PlayerFish(int movespeed, boolean isAlive, String leftImgFileName, String rightImgFileName, Sprite sprite, int score) {
16  		super(movespeed, sprite);
17  		setAlive(isAlive);
18  		setPlayerFishLeftImage(new Image(leftImgFileName));
19  		setPlayerFishRightImage(new Image(rightImgFileName));
20  		this.score=score;
21  	}
22  	
23  	/**
24  	 * This method creates the fish the player controls.
25  	 */
26  	public static PlayerFish createPlayerFish() {
27  		Image playerFishImage = new Image(leftImageName);
28  		
29  		// Create a hitbox for the playerfish. The playerfish will start at the middle of the screen. 
30  		// So the starting position is the respective screen diameters/2. The size of the hitbox is
31  		// the size of the image casted to int values.
32  		AABB aabb = new AABB(MainScreenController.getScreenbox().getWidth()/2, MainScreenController.getScreenbox().getHeight()/2, (int) playerFishImage.getWidth(), (int) playerFishImage.getHeight());
33  		
34  		// Create a new 'sprite' using the image and its corresponding hitbox.
35  		Sprite sprite = new Sprite(playerFishImage, aabb);
36  		
37  		return new PlayerFish(10, true, leftImageName, rightImageName, sprite,0);
38  	}
39  	
40  	/** This method grows the fish when it 'eats' another fish.
41  	 * 
42  	 * @param multiplier, the multiplier for the X and Y values.
43  	 */
44  	public void grow(double multiplier) {
45  		AABB playerFishAABB = this.getSprite().getAabb();
46  		double newWidth = multiplier * this.getSprite().getImg().getWidth();
47  		double newHeight = multiplier * this.getSprite().getImg().getHeight();
48  		
49  		this.setPlayerFishLeftImage(new Image("FishOriginal_transparent.png", newWidth, newHeight, true, true));
50  		this.setPlayerFishRightImage(new Image("Fish_Right_Transparent.png", newWidth, newHeight, true, true));
51  		
52  		playerFishAABB.setWidth((int) this.getPlayerFishLeftImage().getWidth());
53  		playerFishAABB.setHeight((int) this.getPlayerFishLeftImage().getHeight());
54  	}
55  
56  	public boolean playerDies(EnemyFish enemyfish) {
57  		AABB playerAABB = this.getSprite().getAabb();
58  		AABB enemyAABB = enemyfish.getSprite().getAabb();
59  		return((playerAABB.getX() * playerAABB.getY()) <= (enemyAABB.getX() * enemyAABB.getY()));
60  	}

1   package main;
2   import javafx.scene.canvas.GraphicsContext;
3   import javafx.scene.image.Image;
4   
5   /**
6    * A sprite represents the model of an entity on the screen. It has a position in X and Y coordinates,
7    * a certain velocity in a direction expressed by X and Y coordinates, A width and a height, and an image to draw on the screen.
8    * @author Clinton Cao, Dmity Malarev, Matthijs Halvemaan, Sunwei Wang, Michiel Doesburg.
9    */
10  public class Sprite {
11  	private Image img;
12  	private AABB aabb;
13  	
14  	public Sprite(Image img, AABB aabb) {
15  		this.img = img;
16  		setAabb(aabb);	
17  	}
18  	
19  	public void updateX(int x) {
20  		aabb.updateX(x);
21  	}
22  	
23  	public void updateY(int y) {
24  		aabb.updateY(y);
25  	}
26  	
27  	public void render(GraphicsContext gc) {
28  		gc.drawImage(img, aabb.getX(), aabb.getY());
29  	}
30  	
31  	public boolean intersects(Sprite s) {
32  		return s.getAabb().intersects(this.getAabb());
33  	}
34  	
35  //------------Getters and setters-------------------------------------------
36  	public Image getImg() {
37  		return this.img;
38  	}
39  	
40  	public void setImg(Image img) {
41  		this.img = img;
42  	}
43  
44  	public AABB getAabb() {
45  		return aabb;
46  	}
47  
48  	public void setAabb(AABB aabb) {
49  		this.aabb = aabb;
50  	}
51  
52  }
